174  * Средняя (4 балла)
175  *
176  * Для заданного списка пар "акция"-"стоимость" вернуть ассоциативный массив,
177  * содержащий для каждой акции ее усредненную стоимость.
178  *
179  * Например:
180  *   averageStockPrice(listOf("MSFT" to 100.0, "MSFT" to 200.0, "NFLX" to 40.0))
181  *     -> mapOf("MSFT" to 150.0, "NFLX" to 40.0)
182  */
183 fun averageStockPrice(stockPrices: List<Pair<String, Double>>): Map<String, Double> {
184     val result = mutableMapOf<String, Double>()
185     val resultTimes = mutableMapOf<String, Double>()
186     val resultSum = mutableMapOf<String, Double>()
187     for ((a, b) in stockPrices) {
188         if (resultTimes.containsKey(a)) {
189             var k = resultTimes[a]!!
190             k += 1.0
191             resultTimes[a] = k
192             var sum = resultSum[a]!!
193             sum += b
194             resultSum[a] = sum
195         } else {
196             resultTimes[a] = 1.0
197             resultSum[a] = b
198             result[a] = 0.0
199         }
200     }
201     for (element in result) {
202         result[element.key] = resultSum[element.key]!! / resultTimes[element.key]!!
203     }
204     return result.toMap()
205 }

302  * Для заданного списка неотрицательных чисел и числа определить,
303  * есть ли в списке пара чисел таких, что их сумма равна заданному числу.
304  * Если да, верните их индексы в виде Pair<Int, Int>;
305  * если нет, верните пару Pair(-1, -1).
306  *
307  * Индексы в результате должны следовать в порядке (меньший, больший).
308  *
309  * Постарайтесь сделать ваше решение как можно более эффективным,
310  * используя то, что вы узнали в данном уроке.
311  *
312  * Например:
313  *   findSumOfTwo(listOf(1, 2, 3), 4) -> Pair(0, 2)
314  *   findSumOfTwo(listOf(1, 2, 3), 6) -> Pair(-1, -1)
315  */
316 fun findSumOfTwo(list: List<Int>, number: Int): Pair<Int, Int> {
317     val done = mutableMapOf<Int, Int>() // % to index
318     if (number < 0) return Pair(-1, -1)
319     if (number == 0) {
320         val indexOfZero = mutableListOf<Int>()
321         for (i in list.indices) {
322             if (list[i] == 0) indexOfZero.add(i)
323             if (indexOfZero.size == 2) return Pair(indexOfZero[0], indexOfZero[1])
324         }
325     } else {
326         for (i in list.indices) {
327             if (list[i] > number) continue
328             if (list[i] == 0) {
329                 if (done.containsKey(0)) return Pair(
330                     minOf(i, done[0]!!),
331                     maxOf(i, done[0]!!)
332                 )
333                 done[number] = i
334             } else {
335                 if (done.containsKey(number - list[i] % number)) return Pair(
336                     minOf(i, done[number - list[i] % number]!!),
337                     maxOf(i, done[number - list[i] % number]!!)
338                 )
339                 done[list[i] % number] = i
340             }
341         }
342     }
343     return Pair(-1, -1)
344 }
