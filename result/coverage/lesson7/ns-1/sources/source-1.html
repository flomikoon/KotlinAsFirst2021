26 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
27     val writer = File(outputName).bufferedWriter()
28     var currentLineLength = 0
29     fun append(word: String) {
30         if (currentLineLength > 0) {
31             if (word.length + currentLineLength >= lineLength) {
32                 writer.newLine()
33                 currentLineLength = 0
34             } else {
35                 writer.write(" ")
36                 currentLineLength++
37             }
38         }
39         writer.write(word)
40         currentLineLength += word.length
41     }
42     for (line in File(inputName).readLines()) {
43         if (line.isEmpty()) {
44             writer.newLine()
45             if (currentLineLength > 0) {
46                 writer.newLine()
47                 currentLineLength = 0
48             }
49             continue
50         }
51         for (word in line.split(Regex("\\s+"))) {
52             append(word)
53         }
54     }
55     writer.close()
56 }

107                     }
108                 }
109             }

112         }

194 fun alignFileByWidth(inputName: String, outputName: String) {
195     var max = 0
196     val regex = "[ ]+".toRegex()
197     val listFile = mutableListOf<String>()
198     for (line in File(inputName).readLines()) {
199         listFile += line.trim()
200         val thisLine = line.trim().split(regex).joinToString(" ")
201         if (thisLine.length > max) max = thisLine.length
202     }
203     val writer = File(outputName).bufferedWriter()
204     for (line in listFile) {
205         if (line.isEmpty() || regex.matches(line)) writer.newLine()
206         else {
207             if (line.split(regex).size == 1) {
208                 writer.write(line)
209                 writer.newLine()
210             } else {
211                 val thisLine = line.split(regex).toMutableList()
212                 var difference = max - thisLine.joinToString("").length
213                 fun spaceMaker(list: List<String>): List<String> {
214                     for (i in list.indices) {
215                         if (i == list.lastIndex) {
216                             if (difference > 0) spaceMaker(list)
217                             else break
218                         }
219                         if (difference > 0) thisLine[i] += " "
220                         difference -= 1
221                     }
222                     return list
223                 }
224                 writer.write(spaceMaker(thisLine).joinToString(""))
225                 writer.newLine()
226             }
227         }
228     }
229     writer.close()
